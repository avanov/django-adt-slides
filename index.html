let filter_by_oid instructions oid =
List.filter instructions
    (fun x -> match x with
              | Order o -> o.id = oid
              | Cancel c -> c.xid = oid)
;;

(* Two product types *)
type order = { id: int; price: float; size: int; }
type cancel = { xid: int; }

type cancel_replace = { xr_id: int; new_price: float; new_size: int; }
